    9 newMaze :: Integer -> Integer -> [Square] -> Maze
   10 newMaze newRows newColumns newWalls = Maze {grid = thisGrid, walls = newWalls, rows = newRows, columns = newColumns}

   12         thisGrid = [(x, y) | x <- [0 .. newRows - 1], y <- [0 .. newColumns - 1]]
