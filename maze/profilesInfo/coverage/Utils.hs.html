    3 quickSort :: Ord a => [a] -> [a]
    4 quickSort [] = []
    5 quickSort (x:xs) = quickSort [y | y <- xs, y <= x] ++ [x] ++ quickSort [y | y <- xs, y > x]

    7 splitHalf :: [a] -> ([a], [a])
    8 splitHalf xs = (ys, zs)
    9     where
   10         l = length xs
   11         ys = [xs !! i | i <- [0 .. (l - 1)], 2*i < l]
   12         zs = [xs !! i | i <- [0 .. (l - 1)], 2*i >= l]

   24 zipSort :: Ord a => [a] -> [a] -> [a]
